Live Streaming

The player auto-detects live streams. When detected, the progress bar and skip buttons disappear, the speed control is hidden, and an animated LIVE badge appears.

Auto-detection method by protocol

HLS - LEVEL_LOADED event with details.live === true.
DASH - isDynamic() on the manifest.
Native MP4 - infinite/very large duration fallback.

Minimal live stream config
typescript
liveConfig: PlayerConfig = {
  src: 'https://live-stream.m3u8',
  // Live is auto-detected - no extra config needed
  autoplay: true,
  muted: true,        // required for autoplay in most browsers
  enablePiP: true,
};
Reading the live state
typescript
@Component({ ... })
export class MyComponent {
  private ctrl = inject(NgxPlayerControlService);

  get isLive(): boolean {
    return this.ctrl.getState()?.isLive() ?? false;
  }

  get liveText(): string {
    return this.isLive ? '🔴 Live' : 'VOD';
  }
}
Speed control automatically hidden for live streams

When isLive() is true, the playback speed option is removed from the settings menu - changing speed has no effect on a live buffer.

Hot-swap Protocol

Switch between HLS, DASH, MP4, and YouTube at runtime without re-creating the component. The Demo Player section above demonstrates this live.

Via NgxPlayerControlService
typescript
@Component({ ... })
export class ChannelSwitcher {
  private ctrl = inject(NgxPlayerControlService);

  switchTo(url: string): void {
    // Destroys current adapter, resets state,
    // preserves volume + mute, starts new adapter
    this.ctrl.load(url, { autoplay: true }, 'my-player');
  }
}
Via reactive binding
typescript
// Changing [src] or [config] triggers hot-swap automatically
currentSrc = signal('https://stream1.m3u8');

switch(url: string): void {
  this.currentSrc.set(url); // hot-swap fires
}
html
<ngx-sp-player
  [src]="currentSrc()"
  [config]="baseConfig"
  playerId="channel-player">
</ngx-sp-player>
What resets on hot-swap

Playback position, buffer, quality levels, live state, subtitles - all reset. What is preserved: volume level and mute state.

Control each player by ID
typescript
@Component({ ... })
export class WallComponent {
  private ctrl = inject(NgxPlayerControlService);

  muteAll(): void {
    this.cameras.forEach(c =>
      this.ctrl.setMuted(true, c.id)
    );
  }

  pauseAll(): void {
    this.cameras.forEach(c =>
      this.ctrl.pause(c.id)
    );
  }

  focusOn(id: string): void {
    // Pause all except the focused one
    this.cameras.forEach(c => {
      if (c.id !== id) this.ctrl.pause(c.id);
    });
    this.ctrl.play(id);
  }
}
Performance tip: lazy mounting

For large grids, use an IntersectionObserver to only mount players when they enter the viewport. The Playground already does this - players outside the scroll area are replaced with a lightweight placeholder.

Programmatic Control

NgxPlayerControlService lets you control any player from anywhere in your app - a parent component, a shared service, a keyboard shortcut handler.

Inject and control
typescript
import {
  NgxPlayerControlService
} from 'ngx-streaming-player';

@Component({ ... })
export class VideoController {
  private ctrl = inject(NgxPlayerControlService);

  // Playback control
  play()   { this.ctrl.play();   }
  pause()  { this.ctrl.pause();  }
  toggle() {
    const state = this.ctrl.getState();
    state?.isPlaying()
      ? this.ctrl.pause()
      : this.ctrl.play();
  }

  // Navigation
  jumpTo(seconds: number): void {
    this.ctrl.seek(seconds);
  }
  skipForward(): void {
    const t = this.ctrl.getState()?.currentTime() ?? 0;
    this.ctrl.seek(t + 10);
  }

  // Audio
  mute()   { this.ctrl.setMuted(true);  }
  unmute() { this.ctrl.setMuted(false); }
  volume(v: number) { this.ctrl.setVolume(v); }

  // Quality / Speed
  setHD()  { this.ctrl.setQuality('1080p'); }
  halfSpeed() { this.ctrl.setPlaybackRate(0.5); }
}
Read reactive state
typescript
@Component({
  template: `
    <p>Playing: {{ isPlaying() }}</p>
    <p>Progress: {{ progress() | number:'1.0-0' }}%</p>
    <p>Buffered: {{ buffered() | number:'1.0-0' }}%</p>
    <p>Duration: {{ duration() }}s</p>
    <p>Live: {{ isLive() }}</p>
  `
})
export class PlayerInfo {
  private ctrl = inject(NgxPlayerControlService);
  private state = this.ctrl.getState();

  // All signals - update automatically
  isPlaying  = this.state?.isPlaying;
  progress   = this.state?.progress;
  buffered   = this.state?.bufferedPercentage;
  duration   = this.state?.duration;
  isLive     = this.state?.isLive;
  isBuffering= this.state?.isBuffering;
  volume     = this.state?.volume;
}

Custom Theming

Customize every color, border radius, and control size using the theme input or directly with --ngx-sp-* CSS custom properties.

Using the theme input
typescript
playerConfig: PlayerConfig = {
  src: 'https://example.com/video.m3u8',
  theme: {
    primaryColor:    '#7C3AED', // progress bar, active states
    secondaryColor:  '#06B6D4', // secondary accents
    accentColor:     '#A78BFA', // highlights, ripples
    backgroundColor: '#1e1b4b', // player background
    textColor:       '#EDE9FE', // text & icons
    borderRadius:    '6px',     // --ngx-sp-radius
    controlSize:     '40px',    // button hit area
  },
};
Using CSS custom properties
css
/* Scope to a specific player instance */
ngx-sp-player#my-player {
  --ngx-sp-primary:   #7C3AED;
  --ngx-sp-secondary: #06B6D4;

  --ngx-sp-bg-dark:   #1e1b4b;
  --ngx-sp-radius:    6px;

/* Or globally for all players */

  --ngx-sp-primary: #7C3AED;

YouTube Integration

Pass any YouTube URL - the player detects it automatically and loads the IFrame API. No API key required.

Auto-detected by URL
typescript
// All of these are auto-detected as YouTube:
// https://www.youtube.com/watch?v=dQw4w9WgXcQ
// https://youtu.be/dQw4w9WgXcQ
// https://www.youtube.com/embed/dQw4w9WgXcQ

ytConfig: PlayerConfig = {
  src: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ',
  autoplay: false,
  muted: false,
};
Force YouTube protocol
typescript
// Use protocol: 'youtube' to force it (e.g. for IDs)
ytConfig: PlayerConfig = {
  src: 'dQw4w9WgXcQ', // video ID only
  protocol: 'youtube',
  autoplay: false,
};
YouTube API limitations

No PiP - the IFrame API prevents browser Picture-in-Picture. The PiP button is hidden automatically.
Quality - quality levels reported by the IFrame API are suggestions, not guarantees.
Autoplay - YouTube enforces muted autoplay in most browser contexts.

Events & State

React to player events via (ready), (stateChange), (error) outputs, or read reactive signals directly via NgxPlayerControlService.

Output events
html
<ngx-sp-player
  [config]="playerConfig"
  playerId="my-player"
  (ready)="onReady()"
  (stateChange)="onState($event)"
  (error)="onError($event)">
</ngx-sp-player>
typescript
onReady(): void {
  console.log('Player ready');
  this.ctrl.play();
}

onState(state: PlayerState): void {
  if (state.isBuffering) {
    this.showSpinner = true;
  }
  if (state.isLive) {
    this.showLiveBadge = true;
  }
}

onError(error: unknown): void {
  console.error('Playback error:', error);
  this.showFallback = true;
}
Reactive state signals
typescript
@Component({
  // All state updates trigger change detection
  // automatically - no manual subscribe() needed
})
export class MyComp {
  private ctrl = inject(NgxPlayerControlService);
  readonly state = this.ctrl.getState('my-player');

  // Use directly in template
  // state?.isPlaying()        → boolean
  // state?.isLive()           → boolean
  // state?.isBuffering()      → boolean
  // state?.currentTime()      → number (seconds)
  // state?.duration()         → number (seconds, Infinity if live)
  // state?.progress()         → number (0–100)
  // state?.bufferedPercentage()→ number (0–100)
  // state?.volume()           → number (0–1)
  // state?.muted()            → boolean
  // state?.playbackRate()     → number
  // state?.currentQuality()   → string
  // state?.availableQualities()→ QualityLevel[]
  // state?.supportsPiP()      → boolean
  // state?.isPiP()            → boolean
  // state?.isFullscreen()     → boolean

  readonly watchProgress = computed(() => {
    const p = this.state?.progress() ?? 0;
    return `${p.toFixed(0)}% watched`;
  });
}
